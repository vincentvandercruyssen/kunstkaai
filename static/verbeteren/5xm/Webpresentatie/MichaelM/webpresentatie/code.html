// HTML
<button id="actionBtn" type="button">turn on light</button>
<div id="status">De lamp is uit</div>

document.addEventListener('DOMContentLoaded', function(){
  let lampOn = false;
  const btn = document.getElementById('actionBtn');
  btn.addEventListener('click', function(){
    if(lampOn){
      lampOn = false;
    } else {
      lampOn = true;
    }
  
  });
});
